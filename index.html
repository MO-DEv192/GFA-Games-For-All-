// 2D Rocket League-style game implementation let canvas = document.getElementById('gameCanvas'); let ctx = canvas.getContext('2d'); let player = { x: 50, y: canvas.height / 2 - 30, width: 10, height: 60, color: 'blue', score: 0 }; let ai = { x: canvas.width - 60, y: canvas.height / 2 - 30, width: 10, height: 60, color: 'red', score: 0 }; let ball = { x: canvas.width / 2, y: canvas.height / 2, radius: 5, speedX: 5, speedY: 5 }; function draw() { ctx.clearRect(0, 0, canvas.width, canvas.height); ctx.fillStyle = player.color; ctx.fillRect(player.x, player.y, player.width, player.height); ctx.fillStyle = ai.color; ctx.fillRect(ai.x, ai.y, ai.width, ai.height); ctx.fillStyle = 'white'; ctx.beginPath(); ctx.arc(ball.x, ball.y, ball.radius, 0, Math.PI * 2); ctx.fill(); } function update() { ball.x += ball.speedX; ball.y += ball.speedY; // AI movement ai.y += (ball.y - (ai.y + ai.height / 2)) * 0.05; // Ball collision with walls if (ball.y + ball.radius > canvas.height || ball.y - ball.radius < 0) { ball.speedY = -ball.speedY; } // Ball collision with paddles if (ball.x - ball.radius < player.x + player.width && ball.y > player.y && ball.y < player.y + player.height) { ball.speedX = -ball.speedX; } else if (ball.x + ball.radius > ai.x && ball.y > ai.y && ball.y < ai.y + ai.height) { ball.speedX = -ball.speedX; } // Scoring if (ball.x < 0) { ai.score++; resetBall(); } else if (ball.x > canvas.width) { player.score++; resetBall(); } } function resetBall() { ball.x = canvas.width / 2; ball.y = canvas.height / 2; ball.speedX = 5 * (Math.random() > 0.5 ? 1 : -1); ball.speedY = 5 * (Math.random() > 0.5 ? 1 : -1); } function keyDownHandler(e) { if (e.key === 'ArrowUp' && player.y > 0) { player.y -= 20; } else if (e.key === 'ArrowDown' && player.y < canvas.height - player.height) { player.y += 20; } } document.addEventListener('keydown', keyDownHandler); function gameLoop() { draw(); update(); requestAnimationFrame(gameLoop); } requestAnimationFrame(gameLoop); // GFA branding console.log('Game powered by GFA');
